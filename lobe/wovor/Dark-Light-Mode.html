Dark-Light-Mode

By: wovor

Toggle dark and light mode.

let isDarkMode = false;  function toggleTheme() {     const bodyElement = document.body;      if (isDarkMode) {         bodyElement.style.backgroundColor = 'white';         bodyElement.style.color = 'black';     } else {         bodyElement.style.backgroundColor = 'black';         bodyElement.style.color = 'white';     }      isDarkMode = !isDarkMode; }
