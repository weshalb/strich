Scroll to Top

By: wovor

Simple scroll to top js function.

function scrollToTop(scrollDuration) {     const scrollStep = -window.scrollY / (scrollDuration / 15),         scrollInterval = setInterval(function() {             if (window.scrollY != 0) {                 window.scrollBy(0, scrollStep);             } else {                 clearInterval(scrollInterval);             }         }, 15); }  document.getElementById("scrollToTopButton").addEventListener("click", function() {     scrollToTop(500);  // 500 milliseconds to reach the top });
